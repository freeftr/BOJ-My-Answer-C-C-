#include <stdio.h>
int main(){
  int n, k, s[10001]={}, temp;
  scanf("%d %d", &n, &k);
  for(int i=0;i<n;i++){
    scanf("%d", &s[i]);
  }
  for(int i=0;i<n;i++){
    for(int j=0;j<n-1-i;j++){
      if(s[j]>s[j+1]){
        temp=s[j+1];
        s[j+1]=s[j];
        s[j]=temp;
      }
    }
  }
  printf("%d",s[n-k]);
}
